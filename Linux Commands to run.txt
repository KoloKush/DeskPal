**** I haven't done a full test run with these commands on a new Pi, but edits will be made over time****
sudo apt-get update
sudo apt-get -y upgrade
sudo apt-get install python3-pip

sudo apt install --upgrade python3-setuptools

cd ~
pip3 install --upgrade adafruit-python-shell

wget https://raw.githubusercontent.com/adafruit/Raspberry-Pi-Installer-Scripts/master/raspi-blinka.py
sudo -E env PATH=$PATH python3 raspi-blinka.py

pip3 install adafruit-circuitpython-charlcd

pip3 install yfinance
pip3 install cryptocompare

sudo apt-get install libopenblas-dev

mkdir Launch
cd Launch
sudo nano Launcher.sh

"
#!/bin/sh
#launcher.sh
# navigate to home directory then to this directory, then execute python, then back home
cd /
cd home/pi/
python3 Main/Main.py
cd /
"

sudo chmod 755 Launcher.sh
***test the launcher***
sudo sh Launcher.sh

cd
mkdir Logs
crontab -e
@reboot sh /home/pi/Launch/Launcher.sh >/home/pi/Logs/cronlog 2>&1

*****Crontab will log any errors or stasuses into /home/pi/logs/cronlog*****

sudo reboot

*****To access the cronlog:*****
cd Logs
sudo cat cronlog
